001    package lombok.maven;
002    
003    import java.io.File;
004    import java.io.IOException;
005    import java.nio.charset.Charset;
006    import java.nio.charset.UnsupportedCharsetException;
007    import java.util.List;
008    
009    import org.apache.commons.lang3.StringUtils;
010    import org.apache.maven.artifact.Artifact;
011    import org.apache.maven.plugin.AbstractMojo;
012    import org.apache.maven.plugin.MojoExecutionException;
013    import org.apache.maven.plugin.logging.Log;
014    import org.apache.maven.plugins.annotations.Component;
015    import org.apache.maven.plugins.annotations.Parameter;
016    import org.apache.maven.project.MavenProject;
017    
018    import lombok.delombok.Delombok;
019    
020    
021    /**
022     * Abstract mojo to Delombok java source with lombok annotations.
023     *
024     * @author <a href="mailto:anthony@whitford.com">Anthony Whitford</a>
025     * @see <a href="http://projectlombok.org/features/delombok.html">Delombok</a>
026     */
027    public abstract class AbstractDelombokMojo extends AbstractMojo {
028    
029        /**
030         * Specifies whether the delombok generation should be skipped.
031         */
032        @Parameter(property="lombok.delombok.skip", defaultValue="false", required=true)
033        protected boolean skip;
034    
035        /**
036         * Encoding.
037         */
038        @Parameter(property="lombok.encoding", defaultValue="${project.build.sourceEncoding}", required=true)
039        protected String encoding;
040    
041        /**
042         * Verbose flag.  Print the name of each file as it is being delombok-ed.
043         */
044        @Parameter(property="lombok.verbose", defaultValue="false", required=true)
045        protected boolean verbose;
046     
047        /**
048         * Add output directory flag.  Adds the output directory to the Maven build path.
049         */
050        @Parameter(property="lombok.addOutputDirectory", defaultValue="true", required=true)
051        protected boolean addOutputDirectory;
052    
053        /**
054         * The Maven project to act upon.
055         */
056        @Component
057        protected MavenProject project;
058    
059        /**
060         * The plugin dependencies.
061         */
062        @Parameter(property="plugin.artifacts", required=true, readonly=true)
063        private List<Artifact> pluginArtifacts;
064    
065        protected abstract String getGoalDescription ();
066    
067        protected abstract File getOutputDirectory();
068    
069        protected abstract File getSourceDirectory();
070    
071        protected abstract String getSourcePath();
072    
073        protected abstract void addSourceRoot(String path);
074    
075        @Override
076        public void execute() throws MojoExecutionException {
077            final Log logger = getLog();
078            assert null != logger;
079    
080            final String goal = getGoalDescription();
081            logger.debug("Starting " + goal);
082            final File outputDirectory = getOutputDirectory();
083            logger.debug("outputDirectory: " + outputDirectory);
084            final File sourceDirectory = getSourceDirectory();
085            logger.debug("sourceDirectory: " + sourceDirectory);
086            final String sourcePath = getSourcePath();
087            logger.debug("sourcePath: " + sourcePath);
088    
089            if (this.skip) {
090                logger.warn("Skipping " + goal);
091            } else if (sourceDirectory.exists()) {
092                // Build a classPath for delombok...
093                final StringBuilder classPathBuilder = new StringBuilder();
094                for (final Object artifact : project.getArtifacts()) {
095                    classPathBuilder.append(((Artifact)artifact).getFile()).append(File.pathSeparatorChar);
096                }
097                for (final Artifact artifact : pluginArtifacts) {
098                    classPathBuilder.append(artifact.getFile()).append(File.pathSeparatorChar);
099                }
100                final String classPath = classPathBuilder.toString();
101                logger.debug("classpath: " + classPath);
102                final Delombok delombok = new Delombok();
103                delombok.setVerbose(this.verbose);
104                delombok.setClasspath(classPath);
105    
106                if (StringUtils.isNotBlank(this.encoding)) {
107                    try {
108                        delombok.setCharset(this.encoding);
109                    } catch (final UnsupportedCharsetException e) {
110                        logger.error("The encoding parameter is invalid; Please check!", e);
111                        throw new MojoExecutionException("Unknown charset: " + this.encoding, e);
112                    }
113                } else {
114                    logger.warn("No encoding specified; using default: " + Charset.defaultCharset());
115                }
116    
117                try {
118                    delombok.setOutput(outputDirectory);
119                    delombok.setSourcepath(getSourcePath());
120                    delombok.addDirectory(sourceDirectory);
121                    delombok.delombok();
122                    logger.info(goal + " complete.");
123                    
124                    if (this.addOutputDirectory) {
125                        // adding generated sources to Maven project
126                        addSourceRoot(outputDirectory.getCanonicalPath());
127                    }
128                } catch (final IOException e) {
129                    logger.error("Unable to delombok!", e);
130                    throw new MojoExecutionException("I/O problem during delombok", e);
131                }
132            } else {
133                logger.warn("Skipping " + goal + "; no source to process.");
134            }
135        }
136    }
