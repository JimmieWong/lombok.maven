001package lombok.maven;
002
003import java.io.File;
004import java.io.IOException;
005import java.nio.charset.Charset;
006import java.nio.charset.UnsupportedCharsetException;
007import java.util.ArrayList;
008import java.util.List;
009import java.util.Map;
010
011import org.apache.commons.lang3.StringUtils;
012import org.apache.maven.artifact.Artifact;
013import org.apache.maven.plugin.AbstractMojo;
014import org.apache.maven.plugin.MojoExecutionException;
015import org.apache.maven.plugin.logging.Log;
016import org.apache.maven.plugins.annotations.Component;
017import org.apache.maven.plugins.annotations.Parameter;
018import org.apache.maven.project.MavenProject;
019
020import lombok.delombok.Delombok;
021import lombok.delombok.Delombok.InvalidFormatOptionException;
022
023
024/**
025 * Abstract mojo to Delombok java source with lombok annotations.
026 *
027 * @author <a href="mailto:anthony@whitford.com">Anthony Whitford</a>
028 * @see <a href="http://projectlombok.org/features/delombok.html">Delombok</a>
029 */
030public abstract class AbstractDelombokMojo extends AbstractMojo {
031
032    /**
033     * Specifies whether the delombok generation should be skipped.
034     */
035    @Parameter(property="lombok.delombok.skip", defaultValue="false", required=true)
036    protected boolean skip;
037
038    /**
039     * Encoding.
040     */
041    @Parameter(property="lombok.encoding", defaultValue="${project.build.sourceEncoding}", required=true)
042    protected String encoding;
043
044    /**
045     * Verbose flag.  Print the name of each file as it is being delombok-ed.
046     */
047    @Parameter(property="lombok.verbose", defaultValue="false", required=true)
048    protected boolean verbose;
049 
050    /**
051     * Add output directory flag.  Adds the output directory to the Maven build path.
052     */
053    @Parameter(property="lombok.addOutputDirectory", defaultValue="true", required=true)
054    protected boolean addOutputDirectory;
055
056    /**
057     * Formatting preferences.
058     */
059    @Parameter
060    protected Map<String, String> formatPreferences;
061
062    /**
063     * The Maven project to act upon.
064     */
065    @Component
066    protected MavenProject project;
067
068    /**
069     * The plugin dependencies.
070     */
071    @Parameter(property="plugin.artifacts", required=true, readonly=true)
072    private List<Artifact> pluginArtifacts;
073
074    protected abstract String getGoalDescription ();
075
076    protected abstract File getOutputDirectory();
077
078    protected abstract File getSourceDirectory();
079
080    protected abstract String getSourcePath();
081
082    protected abstract void addSourceRoot(String path);
083
084    @Override
085    public void execute() throws MojoExecutionException {
086        final Log logger = getLog();
087        assert null != logger;
088
089        final String goal = getGoalDescription();
090        logger.debug("Starting " + goal);
091        final File outputDirectory = getOutputDirectory();
092        logger.debug("outputDirectory: " + outputDirectory);
093        final File sourceDirectory = getSourceDirectory();
094        logger.debug("sourceDirectory: " + sourceDirectory);
095        final String sourcePath = getSourcePath();
096        logger.debug("sourcePath: " + sourcePath);
097
098        if (this.skip) {
099            logger.warn("Skipping " + goal);
100        } else if (sourceDirectory.exists()) {
101            // Build a classPath for delombok...
102            final StringBuilder classPathBuilder = new StringBuilder();
103            for (final Object artifact : project.getArtifacts()) {
104                classPathBuilder.append(((Artifact)artifact).getFile()).append(File.pathSeparatorChar);
105            }
106            for (final Artifact artifact : pluginArtifacts) {
107                classPathBuilder.append(artifact.getFile()).append(File.pathSeparatorChar);
108            }
109            final String classPath = classPathBuilder.toString();
110            logger.debug("classpath: " + classPath);
111            final Delombok delombok = new Delombok();
112            delombok.setVerbose(this.verbose);
113            delombok.setClasspath(classPath);
114
115            if (StringUtils.isNotBlank(this.encoding)) {
116                try {
117                    delombok.setCharset(this.encoding);
118                } catch (final UnsupportedCharsetException e) {
119                    logger.error("The encoding parameter is invalid; Please check!", e);
120                    throw new MojoExecutionException("Unknown charset: " + this.encoding, e);
121                }
122            } else {
123                logger.warn("No encoding specified; using default: " + Charset.defaultCharset());
124            }
125
126            if (null != formatPreferences && !formatPreferences.isEmpty()) {
127                try {
128                    // Construct a list array just like the command-line option...
129                    final List<String> formatOptions = new ArrayList(formatPreferences.size());
130                    for (final Map.Entry<String, String> entry : formatPreferences.entrySet()) {
131                        final String key = entry.getKey();
132                        // "pretty" is an exception -- it has no value...
133                        formatOptions.add( "pretty".equalsIgnoreCase(key) ? key : (key + ':' + entry.getValue()) );
134                    }
135                    delombok.setFormatPreferences(delombok.formatOptionsToMap(formatOptions));
136                } catch (final InvalidFormatOptionException e) {
137                    logger.error("The formatPreferences parameter is invalid; Please check!", e);
138                    throw new MojoExecutionException("Invalid formatPreferences: " + this.formatPreferences, e);
139                }
140            }
141
142            try {
143                delombok.setOutput(outputDirectory);
144                delombok.setSourcepath(getSourcePath());
145                delombok.addDirectory(sourceDirectory);
146                delombok.delombok();
147                logger.info(goal + " complete.");
148                
149                if (this.addOutputDirectory) {
150                    // adding generated sources to Maven project
151                    addSourceRoot(outputDirectory.getCanonicalPath());
152                }
153            } catch (final IOException e) {
154                logger.error("Unable to delombok!", e);
155                throw new MojoExecutionException("I/O problem during delombok", e);
156            }
157        } else {
158            logger.warn("Skipping " + goal + "; no source to process.");
159        }
160    }
161}
