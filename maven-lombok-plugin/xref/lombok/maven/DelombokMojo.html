26      /**

28       * @parameter expression="${lombok.delombok.skip}" default-value="false"
29       * @required

31      private boolean skip;

33      /**
34       * Encoding.

36       * @required

38      private String encoding;

40      /**
41       * Location of the lombok annotated source files.
42       * @parameter expression="${lombok.sourceDirectory}" default-value="${project.basedir}/src/main/lombok"
43       * @required
44       */
45      private File sourceDirectory;
46  
47      /**
48       * Location of the generated source files.
49       * @parameter expression="${lombok.outputDirectory}" default-value="${project.build.directory}/generated-sources/delombok"
50       * @required
51       */
52      private File outputDirectory;
53  
54      /**
55       * Verbose flag.
56       * @parameter expression="${lombok.verbose}" default-value="false"
57       * @required
58       */
59      private boolean verbose;
60  
61      /**
62       * The Maven project to act upon.
63       * @parameter expression="${project}"
64       * @required
65       * @readonly
66       */
67      private MavenProject project;
68  
69      @Override
70      public void execute() throws MojoExecutionException {
71          final Log logger = getLog();
72          assert null != logger;
73  
74          if (this.skip) {
75              logger.warn("Skipping delombok.");
76          } else if (this.sourceDirectory.exists()) {
77              final Delombok delombok = new Delombok();
78              delombok.setVerbose(this.verbose);
79              try {
80                  delombok.setCharset(this.encoding);
81              } catch (final UnsupportedCharsetException e) {
82                  logger.error("The encoding parameter is invalid; Please check!", e);
83                  throw new MojoExecutionException("Unknown charset: " + this.encoding, e);
84              }
85  
86              try {
87                  delombok.setOutput(this.outputDirectory);
88                  delombok.delombok(this.sourceDirectory);
89                  logger.info("Delombok complete.");
90  
91                  // adding generated sources to Maven project
92                  project.addCompileSourceRoot(outputDirectory.getAbsolutePath());
93  
94              } catch (final IOException e) {
95                  logger.error("Unable to delombok!", e);
96                  throw new MojoExecutionException("I/O problem during delombok", e);
97              }
98          } else {
99              logger.warn("Skipping delombok; no source to process.");
100         }
101     }
