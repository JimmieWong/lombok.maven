77          final Log logger = getLog();
78          assert null != logger;
79  
80          final String goal = getGoalDescription();
81          logger.debug("Starting " + goal);
82          final File outputDirectory = getOutputDirectory();
83          logger.debug("outputDirectory: " + outputDirectory);
84          final File sourceDirectory = getSourceDirectory();
85          logger.debug("sourceDirectory: " + sourceDirectory);
86          final String sourcePath = getSourcePath();
87          logger.debug("sourcePath: " + sourcePath);
88  
89          if (this.skip) {
90              logger.warn("Skipping " + goal);
91          } else if (sourceDirectory.exists()) {
92              // Build a classPath for delombok...
93              final StringBuilder classPathBuilder = new StringBuilder();
94              for (final Object artifact : project.getArtifacts()) {
95                  classPathBuilder.append(((Artifact)artifact).getFile()).append(File.pathSeparatorChar);
96              }
97              for (final Artifact artifact : pluginArtifacts) {
98                  classPathBuilder.append(artifact.getFile()).append(File.pathSeparatorChar);
99              }
100             final String classPath = classPathBuilder.toString();
101             logger.debug("classpath: " + classPath);
102             final Delombok delombok = new Delombok();
103             delombok.setVerbose(this.verbose);
104             delombok.setClasspath(classPath);
105 
106             if (StringUtils.isNotBlank(this.encoding)) {
107                 try {
108                     delombok.setCharset(this.encoding);
109                 } catch (final UnsupportedCharsetException e) {
110                     logger.error("The encoding parameter is invalid; Please check!", e);
111                     throw new MojoExecutionException("Unknown charset: " + this.encoding, e);
112                 }
113             } else {
114                 logger.warn("No encoding specified; using default: " + Charset.defaultCharset());
115             }
116 
117             try {
118                 delombok.setOutput(outputDirectory);
119                 delombok.setSourcepath(getSourcePath());
120                 delombok.addDirectory(sourceDirectory);
121                 delombok.delombok();
122                 logger.info(goal + " complete.");
123                 
124                 if (this.addOutputDirectory) {
125                     // adding generated sources to Maven project
126                     addSourceRoot(outputDirectory.getCanonicalPath());
127                 }
128             } catch (final IOException e) {
129                 logger.error("Unable to delombok!", e);
130                 throw new MojoExecutionException("I/O problem during delombok", e);
131             }
132         } else {
133             logger.warn("Skipping " + goal + "; no source to process.");
134         }
